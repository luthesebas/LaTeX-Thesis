%%%------------------------------------%%%
\newcommand{\questionl}[1]{
	%\refstepcounter{questions} % questions counter referenced with \thequestions
	\phantomsection
	%\addcontentsline{faq}{questions}{\thequestions \quad #1} % Adds question to the list of questions
	\section*{#1}
}
\newcommand{\question}[1] {
	\refstepcounter{questions} % questions counter referenced with \thequestions
	\phantomsection
	\addcontentsline{faq}{questions}{\thequestions \quad #1} % Adds question to the list of questions
	\section*{\namequestion\thequestions\separator #1}
}
